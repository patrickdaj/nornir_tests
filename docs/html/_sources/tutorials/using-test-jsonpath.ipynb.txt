    "import requests\n",
    "from nornir.core.task import Result\n",
    "\n",
    "def get_facts(task, ammount=1, animal_type='cat', server='cat-fact.herokuapp.com'):\n",
    "    result = Result(host=task.host)\n",
    "    url = 'https://{}/facts/random?animal_type={}&amount={}'.format(\n",
    "        server, animal_type, ammount\n",
    "    )\n",
    "    r = requests.get(url)\n",
    "\n",
    "    if r.status_code == 200:\n",
    "        result.failed = False\n",
    "        result.result = r.json()\n",
    "        result.stdout = r.text\n",
    "    else:\n",
    "        result.failed = True\n",
    "        result.stderr = r.text\n",
    "    \n",
    "    return result"
